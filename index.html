Ejemplo codigo Arduino / ESP32
#include <WiFi.h>
#include <HTTPClient.h>
#include <TinyGPS++.h>

const char* serverUrl = "http://tu-servidor/api/km-update";
TinyGPSPlus gps;

void sendKmUpdate(float lat, float lng, long km) {
  HTTPClient http;
  http.begin(serverUrl);
  http.addHeader("Content-Type", "application/json");
  String payload = "{\"vehicle_id\":\"VH-001\",";
  payload += "\"lat\":" + String(lat,6) + ",";
  payload += "\"lng\":" + String(lng,6) + ",";
  payload += "\"km_acumulado\":" + String(km) + "}";
  http.POST(payload);
  http.end();
}

void loop() {
  if (gps.location.isUpdated()) {
    sendKmUpdate(gps.location.lat(), gps.location.lng(),
                 gps.distance.meters()/1000);
  }
  delay(30000);
}
